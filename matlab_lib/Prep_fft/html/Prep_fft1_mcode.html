1function [ prc_data ] = Prep_fft(raw_data, FS, START_F, END_F)
 2
 3% PREPROCESSING Summary of this function goes here
 4% Take the raw data, take fft and output in frequency domain
 5% FS = 192000;
 6L = numel(raw_data);
 7NFFT = 2^nextpow2(L); % Next power of 2 from length of y
 8f = FS/2*linspace(0,1,NFFT/2+1); % frequency band
 9tmp_data = fft(raw_data,NFFT)/L;
10tmp_data = 2*abs(tmp_data(1:NFFT/2+1));
11prc_data = tmp_data(START_F < f & f < END_F);

13
14end
